00468 retval_t pal_timer_stop(uint8_t timer_id)
00469 {
00470     bool timer_stop_request_status = false;
00471     uint8_t curr_index;
00472     uint8_t prev_index;
00473 
00474 
00475     if (timer_id >= TOTAL_NUMBER_OF_TIMERS)
00476     {
00477         return (PAL_TMR_INVALID_ID);
00478     }
00479 
00480     ENTER_CRITICAL_REGION();
00481 
00482     /* Check if any timer has expired. */
00483     internal_timer_handler();
00484 
00485     /* The requested timer is first searched in the running timer queue */
00486     if (running_timers > 0)
00487     {
00488         uint8_t timer_count = running_timers;
00489         prev_index = curr_index = running_timer_queue_head;
00490         while (timer_count > 0)
00491         {
00492             if (timer_id == curr_index)
00493             {
00494                 timer_stop_request_status = true;
00495 
00496                 if (timer_id == running_timer_queue_head)
00497                 {
00498                     running_timer_queue_head =
00499                         timer_array[timer_id].next_timer_in_queue;
00500                     /*
00501                      * The value in OCR corresponds to the timeout pointed
00502                      * by the 'running_timer_queue_head'. As the head has
00503                      * changed here, OCR needs to be loaded by the new
00504                      * timeout value, if any.
00505                      */
00506                     prog_ocr();
00507                 }
00508                 else
00509                 {
00510                     timer_array[prev_index].next_timer_in_queue =
00511                         timer_array[timer_id].next_timer_in_queue;
00512                 }
00513                 /*
00514                  * The next timer element of the stopped timer is updated
00515                  * to its default value.
00516                  */
00517                 timer_array[timer_id].next_timer_in_queue = NO_TIMER;
00518                 break;
00519             }
00520             else
00521             {
00522                 prev_index = curr_index;
00523                 curr_index = timer_array[curr_index].next_timer_in_queue;
00524             }
00525             timer_count--;
00526         }
00527         if (timer_stop_request_status)
00528         {
00529             running_timers--;
00530         }
00531     }
00532 
00533     /*
00534      * The requested timer is not present in the running timer queue.
00535      * It will be now searched in the expired timer queue
00536      */
00537     if (!timer_stop_request_status)
00538     {
00539         prev_index = curr_index = expired_timer_queue_head;
00540         while (NO_TIMER != curr_index)
00541         {
00542             if (timer_id == curr_index)
00543             {
00544                 if (timer_id == expired_timer_queue_head)
00545                 {
00546                     /*
00547                      * The requested timer is the head of the expired timer
00548                      * queue
00549                      */
00550                     if (expired_timer_queue_head == expired_timer_queue_tail)
00551                     {
00552                         /* Only one timer in expired timer queue */
00553                         expired_timer_queue_head = expired_timer_queue_tail =
00554                             NO_TIMER;
00555                     }
00556                     else
00557                     {
00558                         /*
00559                          * The head of the expired timer queue is moved to next
00560                          * timer in the expired timer queue.
00561                          */
00562                         expired_timer_queue_head =
00563                             timer_array[expired_timer_queue_head].next_timer_in_queue;
00564                     }
00565                 }
00566                 else
00567                 {
00568                     /*
00569                      * The requested timer is present in the middle or at the
00570                      * end of the expired timer queue.
00571                      */
00572                     timer_array[prev_index].next_timer_in_queue =
00573                         timer_array[timer_id].next_timer_in_queue;
00574 
00575                     /*
00576                      * If the stopped timer is the one which is at the tail of
00577                      * the expired timer queue, then the tail is updated.
00578                      */
00579                     if (timer_id == expired_timer_queue_tail)
00580                     {
00581                         expired_timer_queue_tail = prev_index;
00582                     }
00583                 }
00584                 timer_stop_request_status = true;
00585                 break;
00586             }
00587             else
00588             {
00589                prev_index = curr_index;
00590                curr_index = timer_array[curr_index].next_timer_in_queue;
00591             }
00592         }
00593     }
00594 
00595     if (timer_stop_request_status)
00596     {
00597         /*
00598          * The requested timer is stopped, hence the structure elements of the
00599          * timer are updated.
00600          */
00601         timer_array[timer_id].timer_cb = NULL;
00602     }
00603 
00604     LEAVE_CRITICAL_REGION();
00605 
00606     if (timer_stop_request_status)
00607     {
00608         return (MAC_SUCCESS);
00609     }
00610 
00611     return (PAL_TMR_NOT_RUNNING);
00612 }

00660 void timer_init(void)
00661 {
00662 #if ((TOTAL_NUMBER_OF_TIMERS > 0) || defined(DOXYGEN))
00663     /*
00664      * Initialize the timer resources like timer arrays
00665      * queues, timer registers
00666      */
00667     uint8_t index;
00668 
00669     running_timers = 0;
00670     timer_trigger = false;
00671 
00672     running_timer_queue_head = NO_TIMER;
00673     expired_timer_queue_head = NO_TIMER;
00674     expired_timer_queue_tail = NO_TIMER;
00675 #ifdef ENABLE_HIGH_PRIO_TMR
00676     high_priority_timer_id = NO_TIMER;
00677 #endif
00678 
00679     for (index = 0; index < TOTAL_NUMBER_OF_TIMERS; index++)
00680     {
00681         timer_array[index].next_timer_in_queue = NO_TIMER;
00682         timer_array[index].timer_cb = NULL;
00683     }
00684 #endif  /* #if ((TOTAL_NUMBER_OF_TIMERS > 0) || defined(DOXYGEN)) */
00685 
00686     /*
00687      * Enable the 32 kHz crystal oscillator. This is required in
00688      * order to allow for the hardware performing a transparent clock
00689      * switch when the main 16 MHz crystal is turned off (i.e., when
00690      * the transceiver module is put to sleep).
00691      */
00692     ASSR = _BV(AS2);
00693 
00694     /*
00695      * Start the MAC symbol counter, resetting the prescaler by the
00696      * same time.  Also enable the rx timestamping feature.
00697      */
00698     SCCR0 = _BV(SCTSE) | _BV(SCEN) | _BV(SCRES);
00699 
00700     /*
00701      * Use SCOCR2 to implement a virtual 28-bit timer. This is needed
00702      * since all layers above PAL operate at a microsecond level, while
00703      * the MAC symbol counter operates at a granularity of MAC symbol
00704      * periods (16 s).  Due to that, the timer values have to be
00705      * scaled at the PAL API border, and this scaling effectively
00706      * reduces the timer span to 28 bits.
00707      *
00708      * Whenever the timer is about to cross the 28-bit boundary, it is
00709      * advanced (by the ISR for SCOCR2) so the hardware will then
00710      * perform a real rollover at the next timer tick.  That way, the
00711      * worst effect from blocking interrupts for too long is that all
00712      * pending timers at the time of rollover will be delayed a bit.
00713      * If the SCOCR2 interrupt is served within 16 s, no additional
00714      * delay will happen.
00715      */
00716     SC_WRITE32(SCOCR2, 0x0FFFFFFFul);
00717     SCIRQS = _BV(IRQSCP2);
00718     SCIRQM |= _BV(IRQMCP2);
00719 }

00784 void  pal_timer_delay(uint16_t delay)
00785 {
00786     bool needs_cli = false;
00787     uint8_t sreg = 0;
00788 
00789     /*
00790      * Any interrupt occurring during the delay calculation will introduce
00791      * additional delay and can also affect the logic of delay calculation.
00792      * Hence the delay implementation is put under critical region as long
00793      * as only a small delay time is requested.
00794      */
00795     if (delay < DELAY_WITHOUT_IRQS_US)
00796     {
00797         needs_cli = true;
00798         sreg = SREG;
00799         cli();
00800     }
00801 
00802     /*
00803      * The logic below depends on the CPU being clocked with 16 MHz,
00804      * and timer 5 being set up for a prescaler of 8, so it will be
00805      * clocked with 2 MHz.  This is the best compromise to achieve the
00806      * 1 s resolutions callers of pal_timer_delay() are expecting.
00807      *
00808      * In case any other CPU clock frequency should be supported that
00809      * cannot work this way, some of the logic below needs to be
00810      * rewritten.  Normally, there should be no real need for a
00811      * different CPU clock frequency, as the ATmega128RFA1 can operate
00812      * at 16 MHz down to the minimal possible supply voltage.
00813      *
00814      * As a special exception, the CPU frequency 15.3846 MHz will be
00815      * tolerated as it is used in some test setups.  This results in
00816      * delays being approximately 4 % too long which is deemed to be
00817      * acceptable for that particular purpose.
00818      */
00819 #if (F_CPU != 16000000UL) && (F_CPU != 15384600UL)
00820 #  error "pal_timer_delay() is only implemented for F_CPU = 16 MHz"
00821 #endif
00822     /*
00823      * With a 2 MHz timer clock, it is not possible to span the entire
00824      * delay range of 65 ms using a single timer 5 cycle, so the delay
00825      * is split into multiple smaller delays if required.
00826      */
00827     while (delay != 0)
00828     {
00829         uint16_t actval;
00830 
00831         /*
00832          * Find whether the entire delay time will fit or not.  Note
00833          * that the 2 MHz figure below must *not* be calculated as
00834          * F_CPU / (1000000 * 8) because that will not work as
00835          * expected in the the 15.3 MHz case.
00836          */
00837         if (delay > UINT16_MAX / 2 /* MHz timer clock */)
00838         {
00839             actval = UINT16_MAX / 2;
00840         }
00841         else
00842         {
00843             actval = delay;
00844         }
00845         delay -= actval;        /* remaining delay for next round */
00846 
00847         /*
00848          * Perform the actual delay.  Setup TCNT5 so it will overflow
00849          * after actval cycles.  Then, clear any pending overflow
00850          * interrupt, and start timer 5 with a prescaler of 8.  Poll
00851          * for the overflow interrupt flag, and stop the timer again
00852          * afterwards.
00853          */
00854         actval *= 2 /* microseconds to timer 5 clock cycles @ 2 MHz */;
00855         TCNT5 = (uint16_t)(-actval);
00856         TIFR5 = _BV(TOV5);
00857         TCCR5B = _BV(CS51);
00858         while ((TIFR5 & _BV(TOV5)) == 0)
00859             /* delay */;
00860         TCCR5B = 0;
00861     }
00862 
00863     if (needs_cli)
00864     {
00865         SREG = sreg;
00866     }

01029 static void start_absolute_timer(uint8_t timer_id,
01030                           uint32_t point_in_time,
01031                           FUNC_PTR handler_cb,
01032                           void *parameter)
01033 {
01034     ENTER_CRITICAL_REGION();
01035 
01036     point_in_time &= MASK_LOW28;
01037 
01038     /* Check is done to see if any timer has expired */
01039     internal_timer_handler();
01040 
01041     if (NO_TIMER == running_timer_queue_head)
01042     {
01043         running_timer_queue_head = timer_id;
01044         timer_array[timer_id].next_timer_in_queue = NO_TIMER;
01045     }
01046     else
01047     {
01048         uint8_t i;
01049         bool timer_inserted = false;
01050         uint8_t curr_index = running_timer_queue_head;
01051         uint8_t prev_index = running_timer_queue_head;
01052 
01053         for (i = 0; i < running_timers; i++)
01054         {
01055             if (NO_TIMER != curr_index)
01056             {
01057                 if (compare_time(timer_array[curr_index].abs_exp_timer,
01058                                      point_in_time))
01059                 {
01060                     /*
01061                      * Requested absolute time value is greater than the time
01062                      * value pointed by the curr_index in the timer array
01063                      */
01064                     prev_index = curr_index;
01065                     curr_index = timer_array[curr_index].next_timer_in_queue;
01066                 }
01067                 else
01068                 {
01069                     timer_array[timer_id].next_timer_in_queue = curr_index;
01070                     if (running_timer_queue_head == curr_index)
01071                     {
01072                         /* Insertion at the head of the timer queue. */
01073                         running_timer_queue_head = timer_id;
01074                     }
01075                     else
01076                     {
01077                         timer_array[prev_index].next_timer_in_queue = timer_id;
01078                     }
01079                     timer_inserted = true;
01080                     break;
01081                 }
01082             }
01083         }
01084         if (!timer_inserted)
01085         {
01086             /* Insertion at the tail of the timer queue. */
01087             timer_array[prev_index].next_timer_in_queue = timer_id;
01088             timer_array[timer_id].next_timer_in_queue = NO_TIMER;
01089         }
01090     }
01091     timer_array[timer_id].abs_exp_timer = point_in_time;
01092     timer_array[timer_id].timer_cb = (FUNC_PTR)handler_cb;
01093     timer_array[timer_id].param_cb = parameter;
01094     running_timers++;
01095 
01096     prog_ocr();
01097 
01098     LEAVE_CRITICAL_REGION();
