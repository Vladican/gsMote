00063 void pal_trx_irq_init(trx_irq_hdlr_idx_t trx_irq_num, FUNC_PTR trx_irq_cb)
00064 {
00065     if (trx_irq_num >= NO_OF_TRX_IRQS)
00066     {
00067         /* Illegal request. */
00068         return;
00069     }
00070     /*
00071      * Set the handler function.
00072      * The handler is set before enabling the interrupt to prepare for spurious
00073      * interrupts, that can pop up the moment they are enabled
00074      */
00075     irq_handler[trx_irq_num] = (irq_handler_t)trx_irq_cb;
00076     ENABLE_TRX_IRQ(trx_irq_num);
00077 }
