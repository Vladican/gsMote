00057 void mcu_clock_init(void)
00058 {
00059     uint8_t low_fuse, prescaler;
00060 
00061     /*
00062      * Determine which oscillator the CPU is configured for, and
00063      * adjust the clock prescaler appropriately.  For the internal RC
00064      * oscillator, the special prescaler value 0x0F must be used to
00065      * adjust the RC oscillator for full 16 MHz operation.  In all
00066      * other cases, value 0x0 is used to use a prescaler of 1.  This
00067      * makes the operation independent of the CKDIV8 fuse setting.
00068      */
00069     ENTER_CRITICAL_REGION();
00070     low_fuse = read_avr_fuse(0);
00071     LEAVE_CRITICAL_REGION();
00072 
00073     if ((low_fuse & 0x0F) == 0x02)
00074     {
00075         /* Internal RC oscillator used. */
00076         prescaler = 0x0F;
00077     }
00078     else
00079     {
00080         /* Any other clock source, usually the 16 MHz crystal oscillator. */
00081         prescaler = clock_div_1;
00082     }
00083     clock_prescale_set(prescaler);
00084 }
