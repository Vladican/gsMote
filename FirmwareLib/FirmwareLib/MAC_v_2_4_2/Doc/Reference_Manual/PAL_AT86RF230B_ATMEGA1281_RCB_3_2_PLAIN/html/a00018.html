00360 retval_t pal_timer_stop(uint8_t timer_id)
00361 {
00362     bool timer_stop_request_status = false;
00363     uint8_t curr_index;
00364     uint8_t prev_index;
00365 
00366 
00367     if (timer_id >= TOTAL_NUMBER_OF_TIMERS)
00368     {
00369         return (PAL_TMR_INVALID_ID);
00370     }
00371 
00372     ENTER_CRITICAL_REGION();
00373 
00374     /* Check if any timer has expired. */
00375     internal_timer_handler();
00376 
00377     /* The requested timer is first searched in the running timer queue */
00378     if (running_timers > 0)
00379     {
00380         uint8_t timer_count = running_timers;
00381         prev_index = curr_index = running_timer_queue_head;
00382         while (timer_count > 0)
00383         {
00384             if (timer_id == curr_index)
00385             {
00386                 timer_stop_request_status = true;
00387 
00388                 if (timer_id == running_timer_queue_head)
00389                 {
00390                     running_timer_queue_head =
00391                         timer_array[timer_id].next_timer_in_queue;
00392                     /*
00393                      * The value in OCR corresponds to the timeout pointed
00394                      * by the 'running_timer_queue_head'. As the head has
00395                      * changed here, OCR needs to be loaded by the new
00396                      * timeout value, if any.
00397                      */
00398                     prog_ocr();
00399                 }
00400                 else
00401                 {
00402                     timer_array[prev_index].next_timer_in_queue =
00403                         timer_array[timer_id].next_timer_in_queue;
00404                 }
00405                 /*
00406                  * The next timer element of the stopped timer is updated
00407                  * to its default value.
00408                  */
00409                 timer_array[timer_id].next_timer_in_queue = NO_TIMER;
00410                 break;
00411             }
00412             else
00413             {
00414                 prev_index = curr_index;
00415                 curr_index = timer_array[curr_index].next_timer_in_queue;
00416             }
00417             timer_count--;
00418         }
00419         if (timer_stop_request_status)
00420         {
00421             running_timers--;
00422         }
00423     }
00424 
00425     /*
00426      * The requested timer is not present in the running timer queue.
00427      * It will be now searched in the expired timer queue
00428      */
00429     if (!timer_stop_request_status)
00430     {
00431         prev_index = curr_index = expired_timer_queue_head;
00432         while (NO_TIMER != curr_index)
00433         {
00434             if (timer_id == curr_index)
00435             {
00436                 if (timer_id == expired_timer_queue_head)
00437                 {
00438                     /*
00439                      * The requested timer is the head of the expired timer
00440                      * queue
00441                      */
00442                     if (expired_timer_queue_head == expired_timer_queue_tail)
00443                     {
00444                         /* Only one timer in expired timer queue */
00445                         expired_timer_queue_head = expired_timer_queue_tail =
00446                             NO_TIMER;
00447                     }
00448                     else
00449                     {
00450                         /*
00451                          * The head of the expired timer queue is moved to next
00452                          * timer in the expired timer queue.
00453                          */
00454                         expired_timer_queue_head =
00455                             timer_array[expired_timer_queue_head].next_timer_in_queue;
00456                     }
00457                 }
00458                 else
00459                 {
00460                     /*
00461                      * The requested timer is present in the middle or at the
00462                      * end of the expired timer queue.
00463                      */
00464                     timer_array[prev_index].next_timer_in_queue =
00465                         timer_array[timer_id].next_timer_in_queue;
00466 
00467                     /*
00468                      * If the stopped timer is the one which is at the tail of
00469                      * the expired timer queue, then the tail is updated.
00470                      */
00471                     if (timer_id == expired_timer_queue_tail)
00472                     {
00473                         expired_timer_queue_tail = prev_index;
00474                     }
00475                 }
00476                 timer_stop_request_status = true;
00477                 break;
00478             }
00479             else
00480             {
00481                prev_index = curr_index;
00482                curr_index = timer_array[curr_index].next_timer_in_queue;
00483             }
00484         }
00485     }
00486 
00487     if (timer_stop_request_status)
00488     {
00489         /*
00490          * The requested timer is stopped, hence the structure elements of the
00491          * timer are updated.
00492          */
00493         timer_array[timer_id].timer_cb = NULL;
00494     }
00495 
00496     LEAVE_CRITICAL_REGION();
00497 
00498     if (timer_stop_request_status)
00499     {
00500         return (MAC_SUCCESS);
00501     }
00502 
00503     return (PAL_TMR_NOT_RUNNING);
00504 }

00631 void  pal_timer_delay(uint16_t delay)
00632 {
00633     /*
00634      * Any interrupt occurring during the delay calculation will introduce
00635      * additional delay and can also affect the logic of delay calculation.
00636      * Hence the delay implementation is put under critical region.
00637      */
00638 
00639     ENTER_CRITICAL_REGION();
00640 
00641     if (delay > MIN_DELAY_VAL)
00642     {
00643         uint16_t target_time = TCNT1 + delay;
00644 
00645         if (target_time > TCNT1)
00646         {
00647             /* Target time is within current span of 16 bit count (0xFFFF) */
00648             while (target_time > TCNT1)
00649             {
00650                 /* No operation, till condition is met */
00651             }
00652         }
00653         else
00654         {
00655             /*
00656              * Target time has rolled over and has gone to next span of 16 bit
00657              * count.
00658              */
00659             while (target_time < TCNT1)
00660             {
00661                 /* No operation, till condition is met */
00662             }
00663             while (target_time > TCNT1)
00664             {
00665                 /* No operation, till condition is met */
00666             }
00667         }
00668     }
00669 
00670     LEAVE_CRITICAL_REGION();
00671 }

00762 static void prog_ocr(void)
00763 {
00764     uint16_t  timeout_high;
00765     uint16_t  timeout_low;
00766     uint32_t  timeout;
00767 
00768     ENTER_CRITICAL_REGION();
00769 
00770     if (NO_TIMER != running_timer_queue_head)
00771     {
00772         timeout = timer_array[running_timer_queue_head].abs_exp_timer;
00773         timeout_high = (uint16_t)(timeout >> SYS_TIME_SHIFT_MASK);
00774 
00775         if (timeout_high == sys_time)
00776         {
00777             timeout_low = (uint16_t)(timeout & HW_TIME_MASK);
00778 
00779             /* Program output compare match */
00780             OCR1A = timeout_low;
00781 
00782             /* Clear pending output compare matches */
00783             TIFR1  = _BV(OCF1A);
00784 
00785             /* Enable output compare match interrupt */
00786             TIMSK1 |= _BV(OCIE1A);
00787         }
00788         else
00789         {
00790             /* Disable output compare match interrupt */
00791             TIMSK1 &= ~_BV(OCIE1A);
00792         }
00793 
00794         uint32_t current_time = gettime();
00795 
00796         /* Safety check: Trigger timer, if next_trigger is in the past. */
00797         if (compare_time(timer_array[running_timer_queue_head].abs_exp_timer, current_time + 1))
00798         {
00799             timer_trigger = true;
00800         }
00801     }
00802

00817 void internal_timer_handler(void)
00818 {
00819     /*
00820      * Flag was set once a timer has expired by the timer ISR or
00821      * by function prog_rc().
00822      */
00823     if (timer_trigger)
00824     {
00825         timer_trigger = false;
00826 
00827         if (running_timers > 0) /* Holds the number of running timers */
00828         {
00829             if ((expired_timer_queue_head == NO_TIMER) &&
00830                 (expired_timer_queue_tail == NO_TIMER))
00831             {
00832                 expired_timer_queue_head = expired_timer_queue_tail =
00833                                                     running_timer_queue_head;
00834             }
00835             else
00836             {
00837                 timer_array[expired_timer_queue_tail].next_timer_in_queue =
00838                                                     running_timer_queue_head;
00839 
00840                 expired_timer_queue_tail = running_timer_queue_head;
00841             }
00842 
00843             running_timer_queue_head =
00844                 timer_array[running_timer_queue_head].next_timer_in_queue;
00845 
00846             timer_array[expired_timer_queue_tail].next_timer_in_queue =
00847                 NO_TIMER;
00848 
00849             running_timers--;
00850 
00851             /*
00852              * As a timer has expired, the OCR1A is programmed (if possible)
00853              * with the new timeout value of the timer pointed by running
00854              * timer queue head
00855              */
00856             prog_ocr();
00857         }
00858     }
00859 }
