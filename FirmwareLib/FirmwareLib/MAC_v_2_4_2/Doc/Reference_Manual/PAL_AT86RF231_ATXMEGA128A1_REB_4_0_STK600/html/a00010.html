00156 static uint8_t eeprom_read_byte(uint8_t *addr)
00157 {
00158     uint16_t addrcpy;
00159 
00160     addrcpy = (uint16_t)addr;
00161 
00162     eeprom_wait_for_NVM();
00163 
00164     NVM.ADDR2 = 0;
00165     NVM.ADDR1 = (addrcpy >> 8) & 0xFF;
00166     NVM.ADDR0 = (addrcpy) & 0xFF;
00167     NVM.CMD =  NVM_CMD_READ_EEPROM_gc;
00168 
00169     /*
00170      * The CMDEX bit must be set within 4 clock cycles after setting the
00171      * protection byte in the CCP register.
00172      */
00173     NVM_EXEC();
00174     return NVM.DATA0;

00210 static void eeprom_write_byte(uint8_t addr, uint8_t value)
00211 {
00212     uint16_t addrcpy;
00213 
00214     addrcpy = (uint16_t)addr;
00215 
00216     /*  Flush buffer to make sure no unintetional data is written and load
00217      *  the "Page Load" command into the command register.
00218      */
00219     eeprom_flush_buffer();
00220     NVM.CMD = NVM_CMD_LOAD_EEPROM_BUFFER_gc;
00221 
00222     /* Set address to write to. */
00223     NVM.ADDR0 = addrcpy & 0xFF;
00224     NVM.ADDR1 = (addrcpy >> 8) & 0x1F;
00225     NVM.ADDR2 = 0x00;
00226 
00227     /* Load data to write, which triggers the loading of EEPROM page buffer. */
00228     NVM.DATA0 = value;
00229 
00230     /*  Issue EEPROM Atomic Write (Erase&Write) command. Load command, write
00231      *  the protection signature and execute command.
00232      */
00233     NVM.CMD = NVM_CMD_ERASE_WRITE_EEPROM_PAGE_gc;
00234     NVM_EXEC();
00235 }
