00200 void pal_trx_read_timestamp(uint32_t *timestamp)
00201 {
00202     /*
00203      * Everytime a transceiver interrupt is triggred, input capture register of
00204      * the AVR is latched. The 'sys_time' is concatenated to the ICR to
00205      * generate the time stamp of the received frame.
00206      * 'sys_time'   'ICR'
00207      *  ---------|--------- => 32 bit timestamp
00208      *   16 bits   16 bits
00209      */
00210     *timestamp = (uint32_t)sys_time << (uint32_t)16;
00211     *timestamp |= (uint32_t)TIME_STAMP_REGISTER;
00212 }

00224 bool pal_calibrate_rc_osc(void)
00225 {
00226     /*
00227      * Use the 32.768 kHz external crystal oscillator as reference clock source.
00228      */
00229     /*
00230      * This is the actual value of the timer to be calibrated
00231      * after each calibration run.
00232      */
00233     uint16_t curr_value_cal_timer = 0;
00234 
00235     /* This is the best OSCCAL value. */
00236     uint8_t best_value_osccal = 0;
00237     /*
00238      * This is the smallest difference between the target value and the actual value
00239      * of the timer the timer to be calibrated.
00240      */
00241 
00242     uint16_t best_dif_timers = ~(uint16_t)0;
00243     /*
00244      * This is the difference between  between the target value and the actual value
00245      * of the timer the timer to be calibrated of the current calibration attempt.
00246      */
00247     uint16_t curr_dif_timers = 0;
00248 
00249     uint16_t counter;
00250     uint8_t tccr2b;
00251     uint8_t tccr1b;
00252     uint8_t tccr1a;
00253     bool cal_result = true;
00254 
00255     ENTER_CRITICAL_REGION();
00256 
00257     /*
00258      * Save current values of timer status.
00259      */
00260     tccr2b = TCCR2B;
00261     tccr1b = TCCR1B;
00262     tccr1a = TCCR1A;
00263 
00264     /*
00265      * Stop timers 1 and 2.
00266      * Set timer 1 to normal mode (no CTC, no PWM, just count).
00267      */
00268     TCCR2B = 0;
00269     TCCR1B = 0;
00270     TCCR1A = 0;
00271 
00272     for (counter = 0; counter < MAX_CAL_LOOP_CNT;  counter++)
00273     {
00274         /*
00275          * Delete pending timer 1 and 2 interrupts, and clear the
00276          * counters.
00277          */
00278         TIFR1 = 0xFF;
00279         TIFR2 = 0xFF;
00280         TCNT2 = 0;
00281         TCNT1 = 0;
00282 
00283         /* Kick timer1 with internal clock source and no prescaler */
00284         TCCR1B = (1 << CS10);
00285 
00286         /*
00287          * Kick timer2 with external 32.768 Hz asynchronous clock
00288          * and no prescaler
00289          */
00290         TCCR2B = (1 << CS20);
00291         ASSR = (1 << AS2);
00292 
00293         /*
00294          * Wait for timer 2 to overflow.
00295          */
00296         while (!(TIFR2 & (1 << TOV2)))
00297         {
00298             /* Wait */
00299         }
00300 
00301 
00302         /*
00303          * Stop timer 1.  Now, TCNT1 contains the number of CPU cycles
00304          * counted during F_CPU / (32 * 256) cycles.
00305          */
00306         TCCR1B = 0;
00307         TCCR2B = 0;
00308 
00309         curr_value_cal_timer = TCNT1;
00310 
00311         /*
00312          * Check if reference timer is running at all,
00313          * i.e. the reference clock is available.
00314          */
00315         if (0 == curr_value_cal_timer)
00316         {
00317             /* Reference timer is not running, return error. */
00318             cal_result = false;
00319             break;
00320         }
00321 
00322         if (curr_value_cal_timer <= (uint16_t)(TARGETVAL_CALIBRATION))
00323         {
00324             curr_dif_timers = (uint16_t)(TARGETVAL_CALIBRATION) - curr_value_cal_timer;
00325         }
00326         else
00327         {
00328             curr_dif_timers = curr_value_cal_timer - (uint16_t)(TARGETVAL_CALIBRATION);
00329         }
00330 
00331         if (curr_dif_timers < best_dif_timers)
00332         {
00333             best_dif_timers = curr_dif_timers;
00334             best_value_osccal = OSCCAL;
00335         }
00336 
00337         if (curr_value_cal_timer <= (uint16_t)(TARGETVAL_CALIBRATION))
00338         {
00339             /* Too slow, increase speed */
00340             OSCCAL++;
00341         }
00342         else
00343         {
00344             /* Too fast, lower speed */
00345             OSCCAL--;
00346         }
00347     }
00348 
00349     TCCR2B = tccr2b;
00350     TCCR1B = tccr1b;
00351     TCCR1A = tccr1a;
00352     LEAVE_CRITICAL_REGION();
00353 
00354     OSCCAL = best_value_osccal;
00355 
00356     return (cal_result);
00357 }
